package latencyprotocol

import (
	"errors"
	"go.dedis.ch/onet/v3/log"
	"net"
	"os"
	"runtime"
	"time"

	"golang.org/x/net/icmp"
	"golang.org/x/net/ipv4"
)

//Ping uses ICMP to ping an IP address
func Ping(srcAddress string, dstAddress string) (time.Duration, error) {
	switch runtime.GOOS {
	case "darwin":
	case "linux":
		log.LLvl1("you may need to adjust the net.ipv4.ping_group_range kernel state")
	default:
		log.LLvl1("not supported on", runtime.GOOS)
		return 0, errors.New("not supported on " + runtime.GOOS)
	}

	c, err := icmp.ListenPacket("udp4", srcAddress)
	if err != nil {
		log.LLvl1("Problem listening for icmp packets")
		return 0, err
	}
	defer c.Close()

	wm := icmp.Message{
		Type: ipv4.ICMPTypeExtendedEchoRequest, Code: 0,
		Body: &icmp.Echo{
			ID: os.Getpid() & 0xffff, Seq: 1,
			Data: []byte("ping latency test"),
		},
	}
	wb, err := wm.Marshal(nil)
	if err != nil {
		log.LLvl1("Problem marshalling message to binary fields")
		return 0, err
	}

	sendingTime := time.Now()
	if _, err := c.WriteTo(wb, &net.UDPAddr{IP: net.ParseIP(dstAddress)}); err != nil {
		log.LLvl1("Problem sending message via icmp")
		return 0, err
	}

	rb := make([]byte, 1500)
	n, peer, err := c.ReadFrom(rb)
	if err != nil {
		log.LLvl1("Problem reading icmp message")
		return 0, err
	}

	latency := time.Now().Sub(sendingTime)
	rm, err := icmp.ParseMessage(58, rb[:n])
	if err != nil {
		log.LLvl1("Problem parsing icmp message")
		return 0, err
	}
	switch rm.Type {
	case ipv4.ICMPTypeExtendedEchoReply:
		log.LLvl1("got reflection from " + peer.String())
		return latency, nil
	default:
		log.LLvl1("got " + string((*rm).Code) + "; want echo reply s")
	}

	return 0, errors.New("Did not receive reply")

}
